type Query {

    # Retrieve a Missionary Profile and associated data by their Missionary ID:
    missionary(missionaryId: ID!): Missionary

    # Retrieve a list of Missionaries based on Missionary Input type using optional filters (default is membership unit number):
    missionaries(
        missionariesInput: MissionariesInput,
        """
        deprecated using unitId: Migrate to missionariesInput using membershipUnitId
        Querying by unitId without missionariesInput may be removed in a future version.
        """
        unitId: ID
    ): [Missionary]

    # Retrieve a list of all possible options related to a specific entity and attribute.
    options(entity: Entity, attributeName: AttributeName): [Option]

    # Retrieve a list of Missionary experiences associated with a member:
    member(cmisUuid: ID!): [Missionary]

    # Retrieve a Leader Profile and associated data by their CMIS ID:
    leader(cmisId: ID!): Leader

    ecclesiasticalUnit(id: ID!): EcclesiasticalUnit

    # Retrieve a proselyting area by ProselytingArea ID:
    proselytingArea(id: ID!): ProselytingArea

    #Retrieve Mission data by Org Number
    mission(id: ID!): Mission

    zone(id: ID!): Zone

    district(id: ID!): District

    # Retrieve a missionary's active assignment by their missionaryNumber/legacyMissId:
    activeAssignment(missionaryNumber: ID!): Assignment

    assignmentLocation(id: ID!): AssignmentLocation

    missionariesByAssignmentLocation(assignmentLocationId: ID!): [Missionary]

    #Retrieve a random set of Missionary ID's for testing the API:
    missionaryDataIds: MissionaryTestDataset

    # Retrieve info on mission boundary adjustments from inq
    missionBoundaryChanges(input: MissionBoundaryChangeInput): [MissionBoundaryChange]
    # Retrieve a list of missionaries with a given unitId
    missionariesByAssignedUnit(unitId: ID!): [Missionary]
    """ List of missionaries by AssignmentLocationId's"""
    missionariesByAssignmentLocations(ids: [ID]): [Missionary]
    """ List of candidates who have started a recommend by their current membership unit number"""
    candidatesByMembershipUnit(id:ID): [Candidate]
    """ List of option values that are returned for smms"""
    smmsOptions: SmmsOptions
    """ Request by UnitID, Returns Proselyting Area configured to receive referrals for that Unit. """
    proselytingAreaForReferral(id: ID): [ProselytingArea]
    """ Gets a list of countries. Can use iso2Codes, iso3Codes, or ids to filter. """
    countries(input: CountriesFilters): [Country]

    """ Retrieve a paginated list of missionaries by Service Department """
    missionariesConnection(
        input: MissionariesConnectionInput,
        first: Int,
        after: String,
        last: Int,
        before: String
    ): MissionariesConnection
}
